def to_b(output_base)
  @degree = 0
  @output_base = output_base

  while (@output_base)**(@degree) < @@value
    @degree += 1
  end
  @degree = @degree - 1

  @coef = Array.new(@degree + 1,0)

  number = @@value
  idx = 0
  while number > 0
    @coef[idx] = (number / ((@output_base)**((@degree-idx).to_i)))
    number = number - @coef[idx]*((@output_base)**(@degree-idx))
    idx += 1
  end

  puts @coef.join(',')
end
