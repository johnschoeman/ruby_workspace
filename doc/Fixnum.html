def to_b(base)
  value = self
  degree = 0

  #find degree(higest power of base for leading digit)
  while value / base**(degree) > 1
  degree += 1
  end

  #popluates an array to house coresponding digits.
  digits = Array.new
  idx = 0
  while idx <= degree
  digit = (value / (base**(degree - idx)))
  digits.unshift(digit)
  value = value - (digit*(base**(degree - idx)))
  idx += 1
  end

  #returns a hase with value arr based on given base and the base itself
  #*note that the order of arr is reversed from the typical format, so that the index value alligns with the degree vaule of the particular digit being considered.
  return Mirrornum.new(digits, base)
end
